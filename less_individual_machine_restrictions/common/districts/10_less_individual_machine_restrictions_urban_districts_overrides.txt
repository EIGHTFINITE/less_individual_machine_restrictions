@base_cost = 300
@city_cost = 500

district_industrial = {
	base_buildtime = 480
	is_capped_by_modifier = no

	prerequisites = {
		tech_basic_industry
	}

	show_on_uncolonized = {
		exists = from
		uses_district_industrial = yes
		NOR = {	
			has_modifier = penal_colony 
			has_modifier = slave_colony 
		}
	}

	potential = {
		exists = owner
		uses_district_industrial = yes
		NOR = {	
			has_modifier = penal_colony 
			has_modifier = slave_colony 
		}
	}

	allow = {
		NOR = {
			has_modifier = resort_colony
			has_modifier = penal_colony
			has_modifier = slave_colony
		}
	}

	conversion_ratio = 1
	convert_to = {
		district_arcology_arms_industry
		district_rw_industrial
		district_prison_industrial
		district_battle_thrall
	}

	resources = {
		category = planet_districts_industrial
		cost = {
			minerals = @city_cost
		}
		upkeep = {
			energy = 2
		}

		upkeep = {
			trigger = {
				exists = owner
				owner = {
					has_edict = industrial_maintenance
				}
			}
			energy = 2
		}
	}

	planet_modifier = {
		planet_housing_add = 2
	}

	triggered_planet_modifier = {
		potential = {
			is_planet_class = pc_machine
			owner = {
				is_machine_empire = yes
				has_ascension_perk = ap_machine_worlds
			}
		}
		modifier = {
			planet_housing_add = 2
		}

	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { is_crafter_empire = yes }
		}
		modifier = {
			planet_max_buildings_add = 0.34
		}
	}

	inline_script = {
		script = jobs/industrial_districts_factory_add
		AMOUNT = 1
	}

	inline_script = {
		script = jobs/industrial_districts_foundry_add
		AMOUNT = 1
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = {
				is_machine_empire = yes
				has_edict = industrial_maintenance
			}
		}
		modifier = {
			job_maintenance_drone_add = 1
		}
	}

	triggered_desc = {
		trigger = {
			exists = owner
			owner = {
				is_machine_empire = yes
				has_edict = industrial_maintenance
			}
		}
		text = job_maintenance_drone_effect_desc
	}
}
